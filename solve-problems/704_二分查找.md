# [二分查找](https://leetcode-cn.com/problems/binary-search/)

- 难度：Easy
- 标签：数组, 二分查找

```ts
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
 function search(nums: number[], target: number): number {
  let left = 0
  let right = nums.length - 1
  while(left <= right) {
    let mid = Math.floor((left + right) / 2)
    if (target === nums[mid]) {
      return mid
    } else if (target > nums[mid]) {
      left = mid + 1
    } else if (target < nums[mid]) {
      right = mid - 1
    }
  }
  return -1
};
```
